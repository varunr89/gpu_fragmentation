% SLIDE 1: Problem -- GPU Fragmentation Wastes Capacity
% ============================================================
\begin{frame}{GPU Clusters Waste Capacity Through Fragmentation}

\begin{columns}[T]
\begin{column}{0.55\textwidth}

\textbf{GPU sharing} splits GPUs across multiple tasks to improve utilization.

\medskip

But it creates \textbf{fragments}: leftover GPU capacity too small for any waiting job.

\bigskip

\statbox{21--42\% of free GPU capacity is unusable}

\bigskip

\begin{itemize}
  \item Average GPU utilization: 25--50\%
  \item Fragments cannot consolidate across GPU boundaries
  \item Equivalent to hundreds of idle GPUs at cluster scale
\end{itemize}

\end{column}
\begin{column}{0.43\textwidth}
  \includegraphics[width=\textwidth]{fig2_topology.png}
\end{column}
\end{columns}

\note{
  \begin{itemize}
    \item GPU clusters for ML training are expensive. Utilization averages 25-50\%.
    \item GPU sharing helps -- multiple tasks on one GPU -- but creates fragments.
    \item Alibaba production data: up to 42\% of free GPUs fragmented and unusable.
    \item This is the equivalent of leaving hundreds of GPUs powered on doing nothing.
    \item TRANSITION: Two recent systems tackle parts of this problem...
  \end{itemize}
}

\end{frame}

  \includegraphics[width=0.95\textwidth]{fig2_gavel_algorithm.png}\\[4pt]
  \small Solves LP to allocate jobs across GPU types.\\
  Reduces JCT by 20--70\%.\\[6pt]
  \textcolor{red}{\textbf{Gap:} Ignores \textit{where} jobs land.}
\end{column}

\begin{column}{0.12\textwidth}
  \centering
  \vspace{2cm}
  \Large $+$\\[8pt]
  \normalsize \textcolor{gray}{?}
\end{column}

\begin{column}{0.44\textwidth}
  \centering
  \textbf{FGD} \textit{(ATC 2023)}\\[4pt]
  \includegraphics[width=0.95\textwidth]{fig3_fgd_algorithm.png}\\[4pt]
  \small Places jobs to minimize fragmentation gradient.\\
  Reduces wasted GPUs by 49\%.\\[6pt]
  \textcolor{red}{\textbf{Gap:} No flexible policy framework.}
\end{column}
\end{columns}

\note{
  \begin{itemize}
    \item Gavel solves an LP to allocate jobs across GPU types based on per-model throughput.
    \item FGD places each job on the node that minimizes fragmentation growth.
    \item They operate at different levels: Gavel picks WHICH GPU type, FGD picks WHICH node.
    \item Natural question: what if we combine them?
    \item TRANSITION: Our approach layers them...
  \end{itemize}
}

% SLIDE 3: Two-Stage Architecture

\begin{frame}{Our Approach: Two-Stage Architecture}

\begin{center}
  \includegraphics[width=0.72\textwidth,height=0.55\textheight,keepaspectratio]{fig1_architecture.png}
\end{center}

\vspace{-4pt}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
  \small\textbf{Stage 1 -- Allocation (Gavel):}\\
  LP computes time fractions per GPU type.\\
  \textit{``Which type, how much?''}
\end{column}
\begin{column}{0.48\textwidth}
  \small\textbf{Stage 2 -- Placement (FGD):}\\
  Selects node minimizing fragmentation gradient.\\
  \textit{``Which specific node?''}
\end{column}
\end{columns}

\note{
  \begin{itemize}
    \item Stage 1: Gavel's LP computes time-fraction allocations per GPU type.
    \item Stage 2: Within the allocated type, FGD selects the node that minimizes fragmentation.
    \item Clean separation -- allocation and placement are orthogonal and compose independently.
    \item Also explored GavelFGD+: feeding placement availability back into LP as POA signal.
    \item TRANSITION: To test this, we needed two very different clusters...
  \end{itemize}
}

\end{frame}

\note{
  \begin{itemize}

    \item On uniform 8-GPU nodes, ALL four strategies produce identical fragmentation.
    \item Why? Any job that fits on one node fits equally well on any other.
    \item On mixed nodes, a 3-GPU job can't use scattered 1-GPU fragments -- placement matters.

% SLIDE 10: Takeaway

\begin{frame}{}

\vfill

\begin{center}

{\Large \textbf{Fragmentation is driven by \textit{node topology diversity},\\[6pt] not GPU type heterogeneity.}}

\vspace{24pt}

\begin{columns}[T]
\begin{column}{0.8\textwidth}
\begin{enumerate}
  \item \textbf{Uniform clusters} don't need fragmentation-aware placement
  \item \textbf{Mixed-hardware clusters} benefit substantially (30--40\% less fragmentation, 17\% more throughput)
  \item \textbf{Allocation and placement are orthogonal} -- compose them independently
\end{enumerate}
\end{column}
\end{columns}
